async function sendChat() { const input = document.getElementById('user-input'); const history = document.getElementById('chat-history'); // V JS používáme .trim() místo .strip()! const val = input.value.trim(); if (!val) return; // Přidání zprávy uživatele do okna history.innerHTML += `
Vy: ${val}
`; input.value = ''; history.scrollTop = history.scrollHeight; try { const res = await fetch('/ai', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ message: val, is_tip: false }) }); const data = await res.json(); // Přidání odpovědi AI history.innerHTML += `
AI: ${data.reply}
`; history.scrollTop = history.scrollHeight; } catch (error) { console.error("Chyba při komunikaci se serverem:", error); history.innerHTML += `
Chyba: Server neodpovídá.
`; } }
