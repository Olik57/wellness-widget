async function sendChat() { console.log("Tlačítko bylo stisknuto!"); // Kontrola v konzoli (F12) const input = document.getElementById('user-input'); const history = document.getElementById('chat-history'); if (!input || !history) { console.error("Chyba: Nepodařilo se najít input nebo history v HTML!"); return; } const val = input.value.trim(); // POZOR: Musí být .trim(), ne .strip() if (!val) return; // Přidání zprávy uživatele do okna history.innerHTML += `
Vy: ${val}
`; input.value = ''; history.scrollTop = history.scrollHeight; try { const res = await fetch('/ai', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ message: val }) }); const data = await res.json(); console.log("Odpověď od serveru:", data); // Přidání odpovědi AI (Klíč musí odpovídat tomu v app.py) if (data.reply) { history.innerHTML += `
AI: ${data.reply}
`; } else { history.innerHTML += `
AI neodpovídá správně.
`; } } catch (error) { console.error("Chyba fetch:", error); history.innerHTML += `
Server je nedostupný.
`; } history.scrollTop = history.scrollHeight; }
